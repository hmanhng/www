#!/bin/bash

ln -sf /usr/share/zoneinfo/Asia/Ho_Chi_Minh /etc/localtime
hwclock --systohc
sed -i '178s/.//' /etc/locale.gen
locale-gen
echo "LANG=en_US.UTF-8" >> /etc/locale.conf
echo "archlinux" >> /etc/hostname
echo "127.0.0.1 localhost" >> /etc/hosts
echo "::1       localhost" >> /etc/hosts
echo "127.0.1.1 arch.localdomain archlinux" >> /etc/hosts

echo root:su | chpasswd
echo -n "Username: " && read name
useradd -m $name
echo $name:su | chpasswd
echo "$name ALL=(ALL) ALL" >> /etc/sudoers.d/$name

sed -i '32s/.//' /etc/pacman.conf
sed -i '33s/.//' /etc/pacman.conf
sed -i '36s/.//' /etc/pacman.conf
sed -i '37s/.//' /etc/pacman.conf
#sed -i '93s/.//' /etc/pacman.conf
#sed -i '94s/.//' /etc/pacman.conf

pacman -Syyu --noconfirm grub os-prober efibootmgr networkmanager git neovim

grub-install --target=x86_64-efi --efi-directory=/boot --bootloader-id=Arch
grub-mkconfig -o /boot/grub/grub.cfg

systemctl enable NetworkManager
